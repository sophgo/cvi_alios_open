## 简介

atempo_icore，即核间音频倍数处理器(inter-core audio atempoer)。 该库运行在应用程序处理器(ap)侧, 实际音频倍数运算工作跑在协处理器(cp)侧。该库屏蔽了底层核间通信的细节，应用开发人员可直接基于该库使用芯片的多核异构运算能力。

    int i;
	float speed = 2.0;
    atempo_icore_t *atempo;
    int fs = 48000, osize;
    int framelen = 512; //samples;
    short dataIn[512], *dataOut;
    FILE *fp_in,*fp_out;
    /* 初始化核间atempo */
    atempo_icore_init();
    /* 申请一个核间atempo实例 */
    atempo = atempo_icore_new(fs);
    if (!atempo)
        return -1;
    atempo_icore_set_speed(atempo, dataIn, dataOut, framelen);

	osize = (1024 + framelen / speed) * sizeof(int16_t);
	dataOut = malloc(osize);

    if((fp_in = fopen("audio.pcm", "rb")) == NULL) {
        printf("open audio in file failed\n");
        return -1;
    }
    if((fp_out = fopen("audio_out.pcm", "wb")) == NULL) {
        printf("open audio out file failed\n");
        return -1;
    }

    while(fread(dataIn, 2, framelen, fp_in) == framelen) {
        /* atempo处理每帧音频数据 */
		atempo_icore_write(atempo, dataIn, framelen);
		rc = atempo_icore_read(atempo, dataOut, osize / sizeof(int16_t));
		if (rc > 0)
			fwrite(dataOut, 2, rc, fp_out);
    }

    /* 释放核间atempo实例, 销毁相关资源 */
    atempo_icore_free(atempo);
    return 0;
}
